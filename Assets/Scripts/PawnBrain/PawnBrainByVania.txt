using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class PawnBrain : MonoBehaviour
{

    [SerializeField]
    private float movementSpeed = 3f;
    [SerializeField]
    private float rotationSpeed = 5f;
    [SerializeField]
    private float reachDistance = 0.1f;

    [SerializeField]
    private ModuleData mainModule;
    [SerializeField]
    private ModuleData targetModule;

    private ModuleData currentModule;
    private ModuleData mainModuleCache;
    private ModulePathFinder pathfinder;
    private List<Vector3> currentPath;
    private int currentPathIndex = 0;
    private bool isMoving = false;
    private Coroutine movementCoroutine;

    [SerializeField]
    private bool drawPath = true;
    [SerializeField]
    private Color pathColor = Color.yellow;

    void Start()
    {
        pathfinder = FindFirstObjectByType<ModulePathFinder>();

        if (mainModule != null)
        {
            currentModule = mainModule;
            if (currentModule.module != null)
            {
                transform.position = currentModule.GetCenterPosition();
            }
        }
    }

    void OnValidate()
    {
        if (mainModule != mainModuleCache)
        {
            mainModuleCache = mainModule;
            currentModule = mainModule;

            if (Application.isPlaying && mainModule != null)
            {
                MoveToModule(mainModule);
            }
        }
    }

    public void MoveToModule(ModuleData targetModule) // имя коллизиться с полем targetModule
    {
        if (isMoving)
        {
            StopMovement();
        }

        if (pathfinder == null || targetModule == null || currentModule == null)
        {
            return;
        }

        if (currentModule == targetModule)
        {
            MoveToPosition(targetModule.GetCenterPosition());
            return;
        }

        Vector3 startPos = transform.position;
        Vector3 targetPos = targetModule.GetCenterPosition();

        currentPath = pathfinder.FindPath(startPos, targetPos);

        if (currentPath != null && currentPath.Count > 1)
        {
            movementCoroutine = StartCoroutine(FollowPath(targetModule));
        }
    }

    public void MoveToPosition(Vector3 targetPosition)
    {
        if (isMoving)
        {
            StopMovement();
        }

        currentPath = new List<Vector3> { transform.position, targetPosition };
        movementCoroutine = StartCoroutine(FollowPath(null));
    }

    public void GoThroughGateway(GateWay gateway)
    {
        if (gateway == null || gateway.connectedGatewayId == -1)
        {
            return;
        }

        GateWay connectedGateway = FindConnectedGateway(gateway);
        if (connectedGateway != null)
        {
            ModuleData nextModule = FindModuleByGateway(connectedGateway);
            if (nextModule != null)
            {
                currentModule = nextModule;

                Vector3 gatewayExit = connectedGateway.GetPosition() + connectedGateway.GetExitDirection() * 1f;
                MoveToPosition(gatewayExit);
            }
        }
    }

    private IEnumerator FollowPath(ModuleData finalTargetModule)
    {
        isMoving = true;
        currentPathIndex = 0;

        while (currentPathIndex < currentPath.Count)
        {
            Vector3 targetPoint = currentPath[currentPathIndex];

            Vector3 direction = (targetPoint - transform.position).normalized;
            if (direction != Vector3.zero)
            {
                Quaternion targetRotation = Quaternion.LookRotation(direction);
                transform.rotation = Quaternion.Slerp(transform.rotation, targetRotation, rotationSpeed * Time.deltaTime);
            }

            while (Vector3.Distance(transform.position, targetPoint) > reachDistance)
            {
                transform.position = Vector3.MoveTowards(transform.position, targetPoint, movementSpeed * Time.deltaTime);
                yield return null;
            }

            CheckForGatewayTransition(targetPoint);

            currentPathIndex++;

            yield return null;
        }

        if (finalTargetModule != null)
        {
            currentModule = finalTargetModule;
            Debug.Log($"Arrived at target module {finalTargetModule.GetHashCode()}");
        }

        isMoving = false;
        movementCoroutine = null;
    }

    private void CheckForGatewayTransition(Vector3 position)
    {
        if (currentModule != null)
        {
            var connectedModules = currentModule.GetConnectedModules();
            for (int i = 0; i < connectedModules.Count; i++)
            {
                GateWay gateway = currentModule.GetGatewayByIndex(i);
                if (gateway != null && Vector3.Distance(position, gateway.GetPosition()) < 0.5f)
                {
                    GoThroughGateway(gateway);
                    break;
                }
            }
        }
    }

    public void StopMovement()
    {
        if (movementCoroutine != null)
        {
            StopCoroutine(movementCoroutine);
            movementCoroutine = null;
        }
        isMoving = false;
        currentPathIndex = 0;
        currentPath?.Clear();
    }

    private GateWay FindConnectedGateway(GateWay gateway)
    {
        var moduleMap = FindFirstObjectByType<ModuleMap>();
        if (moduleMap == null) return null;

        var moduleDataListField = typeof(ModuleMap).GetField("moduleList",
            System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance);

        if (moduleDataListField == null) return null;

        var moduleDataList = moduleDataListField.GetValue(moduleMap) as ModuleDataList;
        return moduleDataList?.GetGatewayById(gateway.connectedGatewayId);
    }

    private ModuleData FindModuleByGateway(GateWay gateway)
    {
        var moduleMap = FindFirstObjectByType<ModuleMap>();
        if (moduleMap == null) return null;

        var moduleDataListField = typeof(ModuleMap).GetField("moduleList",
            System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance);

        if (moduleDataListField == null) return null;

        var moduleDataList = moduleDataListField.GetValue(moduleMap) as ModuleDataList;
        return moduleDataList?.GetModuleByGatewayId(gateway.id);
    }

    public void SetTargetModule(ModuleData module)
    {
        targetModule = module;
    }

    public bool IsMoving()
    {
        return isMoving;
    }

    public ModuleData GetCurrentModule()
    {
        return currentModule;
    }

    void OnDrawGizmos()
    {
        if (!drawPath || currentPath == null || currentPath.Count < 2) return;

        Gizmos.color = pathColor;
        for (int i = 0; i < currentPath.Count - 1; i++)
        {
            Gizmos.DrawLine(currentPath[i], currentPath[i + 1]);
            Gizmos.DrawSphere(currentPath[i], 0.1f);
        }
        Gizmos.DrawSphere(currentPath[currentPath.Count - 1], 0.15f);

        if (isMoving && currentPathIndex < currentPath.Count)
        {
            Gizmos.color = Color.red;
            Gizmos.DrawLine(transform.position, currentPath[currentPathIndex]);
        }
    }
}
